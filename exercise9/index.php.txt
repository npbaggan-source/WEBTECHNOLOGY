<?php
$result = "";
$expression = "";

if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $expression = $_POST['expression'];

    // Replace constants
    $expression = str_replace('pi', pi(), $expression);
    $expression = str_replace('e', exp(1), $expression);

    // Replace functions
    $expression = str_replace('sin(', 'sin(deg2rad(', $expression);
    $expression = str_replace('cos(', 'cos(deg2rad(', $expression);
    $expression = str_replace('tan(', 'tan(deg2rad(', $expression);
    $expression = str_replace('log', 'log10', $expression);
    $expression = str_replace('sqrt', 'sqrt', $expression);
    $expression = str_replace('^', '**', $expression);

    // Validate input (important security layer)
    if (!preg_match('/^[0-9+\-*/().^sincotaqlgrpie ]+$/', $expression)) {
        $result = "Invalid Input";
    } else {
        try {
            $result = eval("return $expression;");
        } catch (Throwable $e) {
            $result = "Error";
        }
    }
}

<script>
function append(value) {
    document.getElementById("display").value += value;
}

function clearDisplay() {
    document.getElementById("display").value = "";
}

function backspace() {
    let display = document.getElementById("display");
    display.value = display.value.slice(0, -1);
}

// Optional: allow Enter key to submit
document.getElementById("display").addEventListener("keypress", function(e) {
    if (e.key === "Enter") {
        e.preventDefault();
        this.form.submit();
    }
});
