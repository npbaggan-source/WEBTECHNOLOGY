<script>
let wishCount = 0;
let cartCount = 0;
let totalPrice = 0;

const wishCountDisplay = document.getElementById("wishCount");
const cartCountDisplay = document.getElementById("cartCount");
const totalPriceDisplay = document.getElementById("totalPrice");
const cartItemsDiv = document.getElementById("cartItems");

// Wishlist
document.querySelectorAll(".wishlist-btn-card").forEach(btn => {
  btn.addEventListener("click", () => {
    wishCount++;
    wishCountDisplay.textContent = wishCount;
  });
});

// Add to Cart
document.querySelectorAll(".add-to-cart-btn").forEach(btn => {
  btn.addEventListener("click", (e) => {

    const card = e.target.closest(".product-cart");
    const name = card.querySelector(".product-name").textContent;
    const price = parseInt(card.querySelector(".product-price").textContent);

    cartCount++;
    totalPrice += price;

    cartCountDisplay.textContent = cartCount;
    totalPriceDisplay.textContent = totalPrice;

    // Show item in cart
    const item = document.createElement("p");
    item.textContent = `${name} - ₹${price}`;
    cartItemsDiv.appendChild(item);
  });
});
</script>
